create database confeitaria default character set utf8 collate utf8_general_ci;
use confeitaria;

create table usuarios(id int(3) not null auto_increment,
nome varchar(30) not null,
role enum("ADM", "Estoquista", "Cliente") not null,
email varchar(255) not null,
cpf varchar(11) not null,
senha varchar(30) not null,
status varchar(50) not null,
unique key cpf (cpf),
primary key(id)) engine=InnoDB DEFAULT CHARSET=utf8;
